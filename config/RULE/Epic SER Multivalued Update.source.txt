
        
            import sailpoint.object.ProvisioningPlan;
            import sailpoint.object.Schema;
            import sailpoint.rule.Account;
            import sailpoint.tools.GeneralException;
            import org.apache.commons.logging.Log;
            import org.apache.commons.logging.LogFactory;

            private List objectToList(Object obj) {
                if (obj instanceof Collection) {
                    return new ArrayList((Collection) obj);
                } else {
                    return new ArrayList(Collections.singletonList(obj));
                }
            }

            ////////////////////////////////////////////////////////////////
            // Main
            ////////////////////////////////////////////////////////////////
            Log log = LogFactory.getLog("openconnector.connector.epicser.MultivalueProvisioningRule");
            String logPrefix = "MultivaluedUpdate - ";
            log.trace(logPrefix + "enter");
            if (plan == null || plan.getAccountRequests() == null) return;
            log.trace(logPrefix + "Account requests size: " + plan.getAccountRequests().size());
            Schema schema = application.getSchema("account");
            if (schema == null) {
                log.warn(logPrefix + "Could not get account schema. Exiting.");
                return;
            }
            for (ProvisioningPlan.AccountRequest accountRequest : plan.getAccountRequests()) {
                if (accountRequest.getOp().equals(ProvisioningPlan.ObjectOperation.Create)) {
                    continue;
                }
                if (accountRequest.getOp().equals(ProvisioningPlan.ObjectOperation.Delete)) {
                    continue;
                }
                if (accountRequest.getAttributeRequests() == null) {
                    continue;
                }
                log.trace(logPrefix + "Attribute requests size: " + accountRequest.getAttributeRequests().size());

                for (ProvisioningPlan.AttributeRequest attributeRequest : accountRequest.getAttributeRequests()) {
                    if (attributeRequest.getOp() == null || attributeRequest.getName() == null || attributeRequest.getValue() == null) {
                        log.warn(logPrefix + "Attribute request is missing op, name, or value..");
                        continue;
                    }

                    log.trace(logPrefix + "attribute request name: " + attributeRequest.getName() +
                        ", value: " + attributeRequest.getValue() + ", operation: " + attributeRequest.getOp());

                    if (!attributeRequest.getOp().equals(ProvisioningPlan.Operation.Add) &&
                        !attributeRequest.getOp().equals(ProvisioningPlan.Operation.Remove)) {
                        continue;
                    }

                    if (!schema.isMultiValued(attributeRequest.getName())) {
                        log.trace(logPrefix + attributeRequest.getName() + " is not multivalued. skipping.");
                        continue;
                    }

                    final String applicationName = accountRequest.getApplicationName() == null ?
                        application.getName() :
                        accountRequest.getApplicationName();
                    if (applicationName == null) {
                        log.error(logPrefix + "ApplicationName not in accountRequest or application object");
                        continue;
                    }
                    log.debug(logPrefix + "Application name: " + applicationName);

                    final String nativeIdentity = accountRequest.getNativeIdentity() == null ?
                        plan.getNativeIdentity() :
                        accountRequest.getNativeIdentity();
                    if (nativeIdentity == null) {
                        log.error(logPrefix + "Native Identity not present in AccountRequest or ProvisioningPlan");
                        continue;
                    }
                    log.debug(logPrefix + "nativeIdentity name: " + nativeIdentity);

                    final Account account = idn.getAccountByNativeIdentity(applicationName, nativeIdentity);
                    if (account == null) {
                        log.error(logPrefix + "Could not find account for app " + applicationName + " and identity " + nativeIdentity);
                        continue;
                    }
                    log.debug(logPrefix + "Found account with uuid: " + account.getUuid());

                    final Object rawExistingVals = idn.getRawAccountAttribute(account, attributeRequest.getName());
                    if (rawExistingVals == null) {
                        log.debug(logPrefix + "Existing account attribute value is null for  " + attributeRequest.getName() + ". Nothing to do.");
                        continue;
                    }

                    log.trace(logPrefix + "Raw existing values class: " + rawExistingVals.getClass());
                    List targetValues = objectToList(rawExistingVals);

                    final Object rawChangeVals = attributeRequest.getValue();
                    log.trace(logPrefix + "Attribute Request values class: " + rawChangeVals.getClass());
                    List changeVals = objectToList(rawChangeVals);

                    if (attributeRequest.getOp().equals(ProvisioningPlan.Operation.Add)) {
                        log.debug(logPrefix + "AttributeRequest with ADD operation found name: " + attributeRequest.getName());
                        targetValues.addAll(changeVals);
                        attributeRequest.setValue(targetValues);
                        log.debug(logPrefix + "Changing operation from \"Add\" to \"Set\"");
                        attributeRequest.setOp(ProvisioningPlan.Operation.Set);
                        log.debug(logPrefix + "Resulting target values: " + targetValues);
                    } else if (attributeRequest.getOp().equals(ProvisioningPlan.Operation.Remove)) {
                        log.debug(logPrefix + "AttributeRequest with REMOVE operation found name: " + attributeRequest.getName());
                        targetValues.removeAll(changeVals);
                        attributeRequest.setValue(targetValues);
                        log.debug(logPrefix + "Changing operation from \"Remove\" to \"Set\"");
                        attributeRequest.setOp(ProvisioningPlan.Operation.Set);
                        log.debug(logPrefix + "Resulting target values: " + targetValues);
                    }
                }
            }
            log.trace(logPrefix + "exit");
